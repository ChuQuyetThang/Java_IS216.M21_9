    public class PhanSo {

        //Tạo biến
        int Tu;
        int Mau;

        // Phương thức:
        public void Nhap() {
            Scanner scanIn = new Scanner(System.in);
            System.out.println("Nhap tu so: ");
            Tu = scanIn.nextInt();
            System.out.println("Nhap mau so: ");
            Mau = scanIn.nextInt();
            while (Mau == 0) {
                System.out.println("Loi!! Nhap lai");
                Mau = scanIn.nextInt();
            }
        }

        public void Xuat() {
            System.out.println(Tu / Mau);
        }

    }

    public int UCLN(int a, int b) {
        a = Math.abs(a);
        b = Math.abs(b);
        while (a != b) {
            if (a > b) {
                a -= b;
            } else {
                b = -a;
            }
        }
        return a;
    }
//Rút gọn phân số    

    public void RutGonPS(PhanSo x) {
        int a = UCLN(x.Tu, x.Mau);
        System.out.println("Phan so sau khi rut gon:");
        System.out.println("Tu so: " + x.Tu / a);
        System.out.println("Mau so: " + x.Mau / a);
    }
//Tính tổng 2 phân số:   
    public void tongPhanSo(PhanSo a, PhanSo b) {
        PhanSo c = null;
        c.Tu = a.Tu * b.Mau + b.Tu * a.Mau;
        c.Mau = a.Mau * b.Mau;
        System.out.println("Tong 2 phan so la: ");
        c.Xuat();
    }
    
//Tính hiệu 2 phân số:      
    public void hieuPhanSo(PhanSo a, PhanSo b) {
        PhanSo c = null;
        c.Tu = a.Tu * b.Mau - b.Tu * a.Mau;
        c.Mau = a.Mau * b.Mau;
        System.out.println("Hieu 2 phan so la: ");
        c.Xuat();
    }
    
//Tính tích 2 phân số:     
    public void tichPhanSo(PhanSo a, PhanSo b) {
        PhanSo c = null;
        c.Tu = a.Tu * b.Tu;
        c.Mau = a.Mau * b.Mau;
        System.out.println("Tich 2 phan so la: ");
        c.Xuat();
    }
    
//Tính thương 2 phân số:    
    public void thuongPhanSo(PhanSo a, PhanSo b) {
        PhanSo c = null;
        c.Tu = a.Tu * b.Mau;
        c.Mau = a.Mau * b.Tu;
        RutGonPS(c);
        System.out.println("Tich 2 phan so la: ");
        c.Xuat();
    }
